{{- /* Emits an with srcset/sizes, intrinsic width/height (no layout shift) and lazy loading for any site image referenced as /images/ (i.e. living in assets/images). Anything else — external URL, SVG/GIF, missing file — falls back to a plain . Args (dict): src (required), alt, title, eager (bool — the first image on a page loads eagerly so a potential LCP image is never lazy). */ -}} {{- $src := .src -}} {{- $alt := .alt | default "" -}} {{- $title := .title -}} {{- $eager := .eager -}} {{- $res := "" -}} {{- if strings.HasPrefix $src "/images/" -}} {{- $res = resources.Get (strings.TrimPrefix "/" $src) -}} {{- if not $res -}} {{- warnf "responsive-image: %q not found under assets/ — emitting a plain that will 404" $src -}} {{- end -}} {{- end -}} {{- if and $res (in (slice "jpeg" "png" "webp" "tiff") $res.MediaType.SubType) -}} {{- /* 1504 = 2x the theme's 47rem content column; nothing wider is ever displayed inline, so larger originals are linked instead. */ -}} {{- $cap := 1504 -}} {{- $variants := slice -}} {{- range $w := slice 480 752 1024 $cap -}} {{- if lt $w $res.Width -}} {{- $variants = $variants | append ($res.Resize (printf "%dx" $w)) -}} {{- end -}} {{- end -}} {{- if le $res.Width $cap -}}{{- $variants = $variants | append $res -}}{{- end -}} {{- $fallback := index $variants 0 -}} {{- range $variants -}}{{- if le .Width 1024 -}}{{- $fallback = . -}}{{- end -}}{{- end -}} {{- $srcset := slice -}} {{- range $variants -}}{{- $srcset = $srcset | append (printf "%s %dw" .RelPermalink .Width) -}}{{- end -}} {{- $full := gt $res.Width $cap -}} {{- if $full }}{{ end -}} {{- if $full }}{{ end -}} {{- else if $res -}} {{- /* Exists as an asset but is not worth resizing (SVG, GIF): publish and serve the original. */ -}} {{- else -}} {{- end -}}
